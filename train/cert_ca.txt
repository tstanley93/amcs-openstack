#####################################################################
#####################  Generate Certs on Laptop   ###################
#####################################################################

###  Generate private key
openssl genrsa -out ~/certs/vnfm.amcs.com.key.pem 2048

###  Create registry.splab.pdsea.f5net.com.cnf
cp ~/certs/registry.splab.pdsea.f5net.com.cnf ~/certs/vnfm.amcs.com.cnf

###  Edit the .cnf
vi ~/certs/vnfm.amcs.com.cnf

### Generate the server cert signing request
openssl req \
-sha256 \
-new \
-key ~/certs/vnfm.amcs.com.key.pem \
-config ~/certs/vnfm.amcs.com.cnf \
-out ~/certs/vnfm.amcs.com.csr

### Verify the CSR
openssl req -text -noout -verify -in ~/certs/vnfm.amcs.com.csr

### Generate the server certificate
openssl \
x509 \
-req \
-extensions v3_req \
-days 3650 \
-sha256 \
-in ~/certs/vnfm.amcs.com.csr \
-CA ~/certs/ca.pem \
-CAkey ~/certs/ca.key.pem \
-CAcreateserial \
-out ~/certs/vnfm.amcs.com.crt \
-extfile ~/certs/vnfm.amcs.com.cnf

### Verify the Certificate
openssl x509 -in ~/certs/vnfm.amcs.com.crt -text -noout

#### Combine Cert and Key
cat ~/certs/vnfm.amcs.com.crt ~/certs/vnfm.amcs.com.key.pem > ~/certs/vnfm.amcs.com.pem

### Check Server
echo quit | openssl s_client -showcerts -connect 10.0.0.128:443
